106     /**
107      * Assert configuration of an attribute is taken into account and has the same value when queried directly or through the MBean.
108      *
109      * @param attributeProperty      the property to use to configure the attribute.
110      * @param attributePropertyValue the value to use for the attribute.
111      * @param attributeName          the name of the attribute in the MBean.
112      * @param attributeGetterName    the name of the getter method for the attribute.
113      * @throws Exception should not happen.
114      */
115     @Test(dataProvider = "testConfigurableAttributeProvider")
116     public void testConfigurableAttribute(String attributeProperty,
117                                           Object attributePropertyValue,
118                                           String attributeName,
119                                           String attributeGetterName) throws Exception {
120 
121         // Create a datasource and set the attribute parameter.
122         final Properties testDataSourceProperties = new Properties();
123         testDataSourceProperties.put(attributeProperty, String.valueOf(attributePropertyValue));
124         final ManagedBasicDataSource testDataSource = (ManagedBasicDataSource) ManagedBasicDataSourceFactory.createDataSource(testDataSourceProperties);
125 
126         // Get the value of the attribute from the MBean.
127         final Object attributeValue = getMBeanServerConnection().getAttribute(new ObjectName(testDataSource.getMBeanName()), attributeName);
128 
129         // Check both values are equal.
130         assertTrue((ManagedBasicDataSource.class.getMethod(attributeGetterName).invoke(testDataSource)).equals(attributeValue));
131     }
